class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int>> v;
        for(int i=0;i<numRows;i++){
            vector<int> v1;
            for(int j=0;j<=i;j++){
                v1.push_back(1);
            }
            v.push_back(v1);
        }
        for(int i=0;i<v.size();i++){
			for(int j=0;j<v[i].size();j++){
				if(j!=0&&j!=i){
					v[i][j]=v[i-1][j-1]+v[i-1][j];
				}
			}
		}
        return v;
    }
};
